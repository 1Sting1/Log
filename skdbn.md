# Interaction Flowchart for the Project

```mermaid

graph TD
    A{Запуск} --> P[Инициализация системы];
    A --> Q{Запись данных в Log}
    P --> C{Проверка статуса RFID}
    C --> B[Создание и отправка UID ключа на сервер];
    B --> Y{Проверка данных с сервера};
    Y --> Q;
    Y -- Валидный --> D[Включение светодиода];
    Y -- Невалидный --> E[Ожидание валидного RFID];
    Y -- Лимит --> C;
    D --> F{Нажата кнопка?};
    F -- Да --> G[Активация насоса];
    G -- Запись времени начала --> Q;
    G --> I[Дозировка жидкости];
    I -- Запись времени окончания --> Q;
    I --> K[Выключение светодиода];
    K --> L[Отправка данных на сервер];
    L --> M[Конец];
    F -- Нет --> N[Ожидание нажатия кнопки];
    E -- Если прошло 5 мин --> C;
    N --> F;
    N -- Если прошло 5 мин --> C;
    M -- Ждём 5 секунд --> C;
    I --> Q;
    K --> Q;
    P --> Q;
    A --> Q;
